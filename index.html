        </div>

    </div>

    <textarea id="editor" spellcheck="false" autocomplete="off" autocorrect="off" autocapitalize="none"></textarea>

    <input type="file" id="fileInput" accept=".txt,.html,.js,.css,.json,.md,.csv">



    <script>

        // DOM Elementleri

        const editor = document.getElementById('editor');

        const fileInput = document.getElementById('fileInput');

        const fileNameDisplay = document.getElementById('fileNameDisplay');

        const btnOpen = document.getElementById('btnOpen');

        const btnSave = document.getElementById('btnSave');

        const btnSaveAs = document.getElementById('btnSaveAs');

        const btnUndo = document.getElementById('btnUndo');

        const btnRedo = document.getElementById('btnRedo');

        

        // Yeni eklenen DOM Elementleri

        const btnSelectAll = document.getElementById('btnSelectAll');

        const btnCut = document.getElementById('btnCut');

        const btnCopy = document.getElementById('btnCopy');

        const btnDelete = document.getElementById('btnDelete');



        // Durum Değişkenleri

        let currentFileName = "İsimsiz.txt";

        let historyStack = [''];

        let historyIndex = 0;

        let isProgrammaticChange = false;



        // Geçmiş (İleri/Geri Alma) İşlemleri

        editor.addEventListener('input', () => {

            if (isProgrammaticChange) {

                isProgrammaticChange = false;

                return;

            }

            if (historyIndex < historyStack.length - 1) {

                historyStack = historyStack.slice(0, historyIndex + 1);

            }

            historyStack.push(editor.value);

            historyIndex++;

        });



        btnUndo.addEventListener('click', () => {

            if (historyIndex > 0) {

                historyIndex--;

                isProgrammaticChange = true;

                editor.value = historyStack[historyIndex];

            }

        });



        btnRedo.addEventListener('click', () => {

            if (historyIndex < historyStack.length - 1) {

                historyIndex++;

                isProgrammaticChange = true;

                editor.value = historyStack[historyIndex];

            }

        });



        // Dosya Açma İşlemi

        btnOpen.addEventListener('click', () => {

            fileInput.click();

        });



        fileInput.addEventListener('change', (event) => {

            const file = event.target.files[0];

            if (!file) return;



            currentFileName = file.name;

            fileNameDisplay.textContent = currentFileName;



            const reader = new FileReader();

            reader.onload = (e) => {

                const content = e.target.result;

                isProgrammaticChange = true;

                editor.value = content;

                

                historyStack = [content];

                historyIndex = 0;

            };

            reader.readAsText(file);

            fileInput.value = '';

        });



        // İndirme (Kaydetme) Fonksiyonu

        function downloadFile(filename, content) {

            const blob = new Blob([content], { type: 'text/plain' });

            const url = URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = filename;

            document.body.appendChild(a);

            a.click();

            setTimeout(() => {

                document.body.removeChild(a);

                URL.revokeObjectURL(url);

            }, 0);

        }



        // Kaydet İşlemi

        btnSave.addEventListener('click', () => {

            downloadFile(currentFileName, editor.value);

        });



        // Farklı Kaydet İşlemi

        btnSaveAs.addEventListener('click', () => {

            const newFileName = prompt("Dosya adını girin (uzantısıyla birlikte, örn: index.html):", currentFileName);

            if (newFileName && newFileName.trim() !== "") {

                currentFileName = newFileName.trim();

                fileNameDisplay.textContent = currentFileName;

                downloadFile(currentFileName, editor.value);

            }

        });



        // Yeni Eklenen İşlevler: Tümünü Seç, Kes, Kopyala, Sil

        btnSelectAll.addEventListener('click', () => {

            editor.focus();

            editor.select();

        });



        btnCopy.addEventListener('click', () => {

            const start = editor.selectionStart;

            const end = editor.selectionEnd;

            const selectedText = editor.value.substring(start, end);

            if (selectedText) {

                navigator.clipboard.writeText(selectedText);

                editor.focus();

            }

        });



        btnCut.addEventListener('click', () => {

            const start = editor.selectionStart;

            const end = editor.selectionEnd;

            const selectedText = editor.value.substring(start, end);

            if (selectedText) {

                navigator.clipboard.writeText(selectedText).then(() => {

                    editor.setRangeText('', start, end, 'end');

                    editor.dispatchEvent(new Event('input')); // Geri alma geçmişini tetikler

                    editor.focus();

                });

            }

        });



        btnDelete.addEventListener('click', () => {

            const start = editor.selectionStart;

            const end = editor.selectionEnd;

            if (start !== end) {

                editor.setRangeText('', start, end, 'end');

                editor.dispatchEvent(new Event('input')); // Geri alma geçmişini tetikler

                editor.focus();

            }

        });



        // Service Worker Kaydı (PWA için)

        if ('serviceWorker' in navigator) {

            window.addEventListener('load', () => {

                navigator.serviceWorker.register('sw.js')

                    .then(registration => console.log('SW kayıtlı:', registration.scope))

                    .catch(error => console.log('SW kayıt hatası:', error));

            });

        }

    </script>
